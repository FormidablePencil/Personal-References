// Return everything with the age over 20. This is with an ternary operator. Linear code.
  return(
    <div className="ninja-list">
      {
        const ninjaList = ninjas.map(ninja => {
          return ninja.age > 20 ? (
            <div className="ninja" key={ninja.id}>
              <div>Name: { ninja.name }</div> 
              <div>Age: { ninja.age }</div> 
              <div>Belt: { ninja.belt } </div> 
            </div>
          ) : null;
        });
      }

// Return everything with the age of 20. We used an if statement here. Easier to grasp.
  const ninjaList = ninjas.map(ninja => {

     } else {

}
